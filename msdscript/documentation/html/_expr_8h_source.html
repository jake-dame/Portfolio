28/******************************************************************************
29 * \class Expr
30 * \brief An abstract, base class representing a mathematical expression.
31 *
32 * The Expression class is an abstract class that defines regular and
33 * virtual functions used to perform various mathematical operations. These
34 * include basic number and variable expressions, as well as operational
35 * expressions such as addition, multiplication, and let substitution. All
36 * classes that inherit from the Expression class are also able to print as a
37 * string in two different styles.
38 *****************************************************************************/
39class Expr
40{
41public:
42
43 /*
44 * Non-virtual methods
45 */
46 std::string to_string();
47
48 std::string to_pretty_string();
49
50 /*
51 * Pure virtual methods
52 */
53 virtual bool equals( Expr * e ) = 0;
54
55 virtual Val * interp() = 0;
56
57 virtual bool has_variable() = 0;
58
59 virtual Expr * subst( std::string str, Expr * e ) = 0;
60
61 virtual void print( std::ostream & stream ) = 0;
62
63 /*
64 * Regular virtual methods
65 */
66 virtual void pretty_print( std::ostream & stream )
67 {
68 print( stream );
69 }
70
71 virtual void pretty_print_at( std::ostream & stream,
72 prec_t p,
73 std::streampos & pos,
74 bool paren )
75 {
76 pretty_print( stream );
77 }
78};
